/* PAGE SWITCH */

function showPage(page){

}

/* STAR MESSAGES */

function showStar(n){

let messages=[

"You're a gentleman",

"You never raise your voice at me",

"You always listen to me",

"You treat me like I'm everything",

"You broke my standards chart"

]

let box=document.getElementById("starMessage")

box.style.display="block"

box.innerText=messages[n-1]

}

/* CREATE STARS ONLY ON STAR PAGE */

function createStars(){

let starPage=document.getElementById("stars")

for(let i=0;i<70;i++){

let star=document.createElement("div")

star.className="bgStar"

star.innerHTML="✦"

star.style.left=Math.random()*100+"vw"

star.style.top=Math.random()*100+"vh"

starPage.appendChild(star)

}

}

/* PETALS */

function petals(){

let icons=["🌸","✨","💗"]

let el=document.createElement("div")

el.className="petal"

el.innerHTML=icons[Math.floor(Math.random()*icons.length)]

el.style.left=Math.random()*100+"vw"

el.style.animationDuration=5+Math.random()*5+"s"

document.body.appendChild(el)

setTimeout(()=>{el.remove()},10000)

}

setInterval(petals,700)
